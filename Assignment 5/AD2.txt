Node* reverseKGroup(Node* head, int k) {
    if (head == NULL) {
        return NULL;
    }

    Node* checker = head;
    for (int i = 0; i < k; i++) {
        if (checker == NULL) {
            return head;
        }
        checker = checker->next;
    }

    Node* prev = NULL;
    Node* current = head;
    Node* nextNode = NULL;
    int count = 0;

    while (current != NULL && count < k) {
        nextNode = current->next;
        current->next = prev;
        prev = current;
        current = nextNode;
        count++;
    }

    if (nextNode != NULL) {
        head->next = reverseKGroup(nextNode, k);
    }

    return prev;
}
